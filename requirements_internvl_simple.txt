#      #   - echo $CUDA_HOME  # Should show CUDA path (may be empty if CUDA is in system PATH)
#      pip install flash-attn==2.5.7 --no-build-isolation
#      # Alternative if above fails, try:
#      #   pip install flash-attn==2.5.7 --no-build-isolation --no-deps
#      # Note: This compilation may take 10-15 minutes. Be patient!
#      # If compilation fails, check:
#      #   - Is g++ available? (run: g++ --version)
#      #   - Is CUDA_HOME set? (run: echo $CUDA_HOME)
#      #   - Is nvcc available? (run: nvcc --version)
