Installation (MUST follow this order):
  1. Create conda environment:
     conda create -n internvl python=3.10 -y
     conda activate internvl

  2. Load compiler and CUDA toolchains (REQUIRED for flash-attn compilation):
     # The system uses "blindfold" to hide system compilers, so you MUST load via spack
     # If using spack (recommended):
     spack load /lhqcen5                 # GCC toolchain (provides g++, gcc, gfortran)
     # Note: If you see "gcc is already loaded", that's fine - continue to next step
     spack load cuda@12.4.0/obxqih4      # CUDA toolkit (provides nvcc, headers, libs)
     
     # Set CUDA environment variables:
     export CUDA_HOME=$(spack location -i cuda@12.4.0/obxqih4)
     export PATH=$CUDA_HOME/bin:$PATH
     export LD_LIBRARY_PATH=$CUDA_HOME/lib64:$LD_LIBRARY_PATH
     
     # IMPORTANT: Explicitly add GCC bin to PATH (blindfold may still hide it):
     export GCC_HOME=$(spack location -i /lhqcen5)
     export PATH=$GCC_HOME/bin:$PATH
     
     # Verify compiler and CUDA are available:
     g++ --version  # Should show "g++ (Spack GCC) 12.4.1", NOT "blindfold" error
     nvcc --version  # Should show CUDA compiler version
     echo "CUDA_HOME: $CUDA_HOME"  # Should show CUDA installation path
     # If /lhqcen5 is not available, try: spack load gcc@12.3.1/<hash>
     #   (find hash with: spack find -l gcc@12.3.1)

  3. Install build dependencies for flash-attn:
     pip install psutil packaging ninja

  4. Install PyTorch with CUDA support FIRST (required for flash-attn):
     pip install torch==2.4.0 torchvision==0.19.0 torchaudio==2.4.0 --index-url https://download.pytorch.org/whl/cu121

  5. Install flash-attn (requires PyTorch, CUDA, compiler, and build dependencies):
     # IMPORTANT: Ensure compiler and CUDA are loaded (step 2) before this step!
     # Verify environment is set up correctly:
     #   - g++ --version  # Should show "g++ (Spack GCC) 12.4.1", NOT "blindfold" error
     #   - nvcc --version  # Should show CUDA compiler version
     #   - echo $CUDA_HOME  # Should show CUDA installation path
     #   - echo $GCC_HOME  # Should show GCC installation path
     pip install flash-attn==2.5.7 --no-build-isolation
     # Note: This compilation may take 10-15 minutes. Be patient!
     # If compilation fails with "g++ not found" or "blindfold" error:
     #   - Re-run step 2 to ensure GCC_HOME is in PATH
     #   - Check: which g++  # Should point to spack GCC, not /tools/bin/blindfold/g++
     # If compilation fails with "CUDA_HOME not set":
     #   - Re-run step 2 to ensure CUDA_HOME is set
     #   - Check: nvcc --version  # Should work
     # You can skip flash-attn if it fails (it's optional for some use cases, but recommended)

  6. Install other requirements:
     pip install -r requirements_internvl.txt
